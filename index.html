    <script>

        const canvas = document.getElementById("renderCanvas");

        const engine = new BABYLON.Engine(canvas, true);

        let xpCount = 0;


        const createScene = () => {

            const scene = new BABYLON.Scene(engine);

            scene.clearColor = new BABYLON.Color3(0.4, 0.6, 0.9); // Dunedin Sky Blue


            // Camera setup (First Person View)

            const camera = new BABYLON.FreeCamera("camera", new BABYLON.Vector3(0, 2, -10), scene);

            camera.attachControl(canvas, true);

            camera.applyGravity = true;

            camera.ellipsoid = new BABYLON.Vector3(1, 1, 1);

            camera.checkCollisions = true;


            const light = new BABYLON.HemisphericLight("light", new BABYLON.Vector3(0, 1, 0), scene);


            // Create the Ground (The Quad)

            const ground = BABYLON.MeshBuilder.CreateGround("ground", {width: 200, height: 200}, scene);

            const groundMat = new BABYLON.StandardMaterial("gMat", scene);

            groundMat.diffuseColor = new BABYLON.Color3(0.1, 0.4, 0.1); // Grass

            ground.material = groundMat;

            ground.checkCollisions = true;


            // Simple Minecraft-style Blocky Building (The Clocktower)

            const tower = BABYLON.MeshBuilder.CreateBox("tower", {height: 15, width: 4, depth: 4}, scene);

            tower.position = new BABYLON.Vector3(20, 7.5, 20);

            const towerMat = new BABYLON.StandardMaterial("tMat", scene);

            towerMat.diffuseColor = new BABYLON.Color3(0.4, 0.4, 0.4); // Stone Grey

            tower.material = towerMat;


            // The "Leith River" (Blue Blocks)

            const river = BABYLON.MeshBuilder.CreateBox("river", {height: 0.1, width: 10, depth: 100}, scene);

            river.position = new BABYLON.Vector3(-15, 0.05, 0);

            const riverMat = new BABYLON.StandardMaterial("rMat", scene);

            riverMat.diffuseColor = new BABYLON.Color3(0, 0.3, 0.8);

            river.material = riverMat;


            // MISSION LOGIC: Check if player is near the tower

            scene.onBeforeRenderObservable.add(() => {

                let dist = BABYLON.Vector3.Distance(camera.position, tower.position);

                if (dist < 5 && xpCount === 0) {

                    xpCount = 500;

                    document.getElementById("xp").innerText = xpCount;

                    document.getElementById("rankXp").innerText = xpCount;

                    alert("Mission Success! You reached the Clocktower. +500 XP");

                }

            });


            return scene;

        };


        const scene = createScene();

        engine.runRenderLoop(() => scene.render());

        window.addEventListener("resize", () => engine.resize());
